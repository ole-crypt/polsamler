function genVolHist(days,seed){const r=seededRand(seed+999);return Array.from({length:days},()=>Math.floor(r()*900+100));}

function genDates(days){return Array.from({length:days},(_,i)=>{const d=new Date();d.setDate(d.getDate()-(days-1-i));return d.toLocaleDateString('no',{day:'2-digit',month:'short'});});}

function saveObs(p){const key='obs_'+p.name.toLowerCase().replace(/\s+/g,'_');let obs=[];try{obs=JSON.parse(localStorage.getItem(key)||'[]');}catch(e){}const today=new Date().toISOString().split('T')[0];if(!obs.find(o=>o.date===today)){obs.push({date:today,price_no:p.price_no,price_se:p.price_se});if(obs.length>365)obs=obs.slice(-365);try{localStorage.setItem(key,JSON.stringify(obs));}catch(e){}}}

// SEARCH

async function doSearch(){

  const q=document.getElementById('searchInput').value.trim();if(!q)return;

  document.getElementById('results').innerHTML=`<div class="loading"><div class="spinner"></div><br>Henter priser…</div>`;

  try {

    const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`);

    if(!res.ok) throw new Error('API-feil');

    const {no,se} = await res.json();

    appState.results = [...no,...se];

    appState.isDemo = appState.results.length===0;

    if(appState.isDemo){

      const lq=q.toLowerCase();

      appState.results=DEMO.filter(p=>p.name.toLowerCase().includes(lq)||p.category.includes(lq));

    }

  } catch(e) {

    const lq=q.toLowerCase();

    appState.results=DEMO.filter(p=>p.name.toLowerCase().includes(lq)||p.category.includes(lq));

    appState.isDemo=true;

  }

  applyFilter();

}


function applyFilter(){

  let data=appState.results.filter(p=>!appState.activeCategory||p.category===appState.activeCategory);

  data.forEach(p=>{p.price_nok=p.source==='no'?p.price:p.price/RATE;p.price_sek=p.source==='se'?p.price:p.price*RATE;p.mlpkr=p.vol&&p.price>0?p.vol/p.price:0;});

  data.sort((a,b)=>{let av=a[appState.sortCol],bv=b[appState.sortCol];if(typeof av==='string'){av=av.toLowerCase();bv=bv.toLowerCase();}return appState.sortDir==='asc'?(av>bv?1:-1):(av<bv?1:-1);});

  appState.filtered=data;renderTable();

}


function renderTable(){

  seItems.filter(s=>!usedSe.has(s.id)).forEach(se=>rows.push({no:null,se}));

}
